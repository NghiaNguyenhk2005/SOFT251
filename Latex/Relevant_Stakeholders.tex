\section{Các bên liên quan (Relevant Stakeholders)}
\begin{center}
\begin{table}[H]
\begin{tabular}{|p{4.5cm}|p{10cm}|}
\hline
\textbf{Nhóm Stakeholder} & \textbf{Vai trò và yêu cầu} \\
\hline
\textbf{Sinh viên (Mentee)} & Đăng ký tham gia chương trình, lựa chọn hoặc được gợi ý tutor phù hợp, đặt lịch tư vấn, phản hồi chất lượng buổi học, nhận hỗ trợ học tập cá nhân hóa. \\
\hline
\textbf{Tutor (Giảng viên, Nghiên cứu sinh, Sinh viên năm trên)} & Thiết lập lịch rảnh, mở buổi tư vấn, quản lý buổi gặp, ghi nhận tiến bộ mentee, chia sẻ tài liệu học tập, nhận phản hồi từ mentee. \\
\hline
\textbf{Khoa / Bộ môn} & Khai thác dữ liệu đánh giá để nắm tình hình học tập của sinh viên, hỗ trợ phân bổ nguồn lực giảng dạy. \\
\hline
\textbf{Phòng Đào tạo (PĐT)} & Sử dụng báo cáo tổng quan để tối ưu hóa phân bổ nguồn lực, theo dõi hiệu quả chương trình. \\
\hline
\textbf{Phòng Công tác Sinh viên (P.CTSV)} & Căn cứ vào kết quả tham gia để cộng điểm rèn luyện, xét học bổng, hỗ trợ phát triển kỹ năng mềm. \\
\hline
\textbf{Quản trị viên hệ thống} & Thiết kế, xây dựng, bảo trì hệ thống, tích hợp các dịch vụ nội bộ, triển khai các tính năng nâng cao như AI ghép cặp, cộng đồng trực tuyến. \\
\hline
\textbf{Điều phối viên} & Quản lý vận hành chương trình, phân bổ tutor, theo dõi tiến độ, xử lý các vấn đề phát sinh, tổng hợp báo cáo. \\
\hline
\textbf{Trung tâm xác thực (HCMUT\_SSO)} & Cung cấp dịch vụ xác thực tập trung. \\
\hline
\textbf{Trung tâm dữ liệu (HCMUT\_DATACORE)} & Đồng bộ dữ liệu cá nhân. \\
\hline
\textbf{Thư viện HCMUT (HCMUT\_LIBRARY)} & Cung cấp tài nguyên học tập, hỗ trợ chia sẻ tài liệu giữa tutor và mentee. \\
\hline
\end{tabular}
\end{table}
\end{center}
